<script>
  // Animationen initialisieren
  document.addEventListener('DOMContentLoaded', function() {
    // Page Load Animationen
    setTimeout(function() {
      document.querySelectorAll('[data-load]').forEach(function(el) {
        el.classList.add('is-loaded');
      });
    }, 150);
    
    // Scroll Animationen
    var observer = new IntersectionObserver(function(entries) {
      entries.forEach(function(entry) {
        if (entry.isIntersecting) {
          var target = entry.target;
          
          if (target.hasAttribute('data-stagger')) {
            var children = Array.from(target.children);
            children.forEach(function(child, index) {
              setTimeout(function() {
                child.classList.add('is-visible');
              }, index * 120);
            });
          } else {
            target.classList.add('is-visible');
          }
          observer.unobserve(target);
        }
      });
    }, { threshold: 0.1, rootMargin: '0px 0px -10% 0px' });
    
    document.querySelectorAll('[data-animate], [data-stagger]').forEach(function(el) {
      observer.observe(el);
    });
  });
</script>
