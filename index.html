Installation Code

<style>
  :root {
    --ease-premium: cubic-bezier(0.16, 1, 0.3, 1);
    --duration-premium: 1.2s;
  }
  
  [data-animate], [data-load], [data-stagger] > * {
    opacity: 0;
    will-change: translate, rotate, scale, opacity, filter;
    transition: opacity var(--duration-premium) var(--ease-premium),
                translate var(--duration-premium) var(--ease-premium),
                rotate var(--duration-premium) var(--ease-premium),
                scale var(--duration-premium) var(--ease-premium),
                filter var(--duration-premium) var(--ease-premium);
  }
  
  [data-animate="up"], [data-load="up"], [data-stagger="up"] > * { translate: 0 40px; }
  [data-animate="down"], [data-load="down"] { translate: 0 -40px; }
  [data-animate="left"] { translate: 40px 0; }
  [data-animate="right"] { translate: -40px 0; }
  [data-animate="zoom"], [data-load="zoom"], [data-stagger="zoom"] > * { scale: 0.94; }
  [data-animate="fade"] { translate: 0 0; }
  [data-animate="blur"] { filter: blur(12px); }
  [data-animate="rotate"] { rotate: -6deg; scale: 0.94; }
  [data-animate="3d"] { transform: perspective(1000px) rotateX(20deg) translateY(40px); }
  
  .is-visible, .is-loaded {
    opacity: 1 !important; translate: 0 0; scale: 1; rotate: 0deg; filter: blur(0);
    transform: perspective(1000px) rotateX(0) translateY(0) !important;
  }
  
  [data-delay="100"] { transition-delay: 100ms; }
  [data-delay="200"] { transition-delay: 200ms; }
  [data-delay="300"] { transition-delay: 300ms; }
  [data-delay="400"] { transition-delay: 400ms; }
  [data-delay="500"] { transition-delay: 500ms; }
</style>

<script>
document.addEventListener('DOMContentLoaded', () => {
    setTimeout(() => {
        document.querySelectorAll('[data-load]').forEach(el => el.classList.add('is-loaded'));
    }, 150);
    const observer = new IntersectionObserver(entries => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                const target = entry.target;
                if (target.hasAttribute('data-stagger')) {
                    Array.from(target.children).forEach((child, i) => {
                        setTimeout(() => child.classList.add('is-visible'), i * 120);
                    });
                } else {
                    target.classList.add('is-visible');
                }
                observer.unobserve(target);
            }
        });
    }, { threshold: 0.1, rootMargin: '0px 0px -10% 0px' });
    document.querySelectorAll('[data-animate], [data-stagger]').forEach(el => observer.observe(el));
});
</script>
